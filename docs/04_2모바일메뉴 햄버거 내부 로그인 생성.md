```html

<div class="collapse navbar-collapse" id="navbarSupportedContent">
    <!-- 모바일 전용 로그인 메뉴 -->

        ddd
    </div>
    <!-- navbar 메뉴 -->
    <ul class="navbar-nav me-auto mb-2 mb-lg-0 mx-0">
```
